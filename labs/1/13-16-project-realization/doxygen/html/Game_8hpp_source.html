149 template<typename T>
150 T* Game::getClosest(const std::vector<T*> objects, Direction dir)
151 {
152  // New: Method used in collisions
153  if (objects.empty())
154  return nullptr;
155 
156  // std::function can be thought of as something similar to "pointer to a function",
157  // even though it's quite a bit more complex
158  std::function<bool(const T*, const T*)> comparator;
159 
160  // Select the correct comparator function
161  switch (dir)
162  {
163  case Direction::DOWN:
164  comparator = [](const T* obj1, const T* obj2) { return obj1->getPosition().y < obj2->getPosition().y; };
165  break;
166  case Direction::UP:
167  comparator = [](const T* obj1, const T* obj2) { return obj1->getPosition().y > obj2->getPosition().y; };
168  break;
169  case Direction::RIGHT:
170  comparator = [](const T* obj1, const T* obj2) { return obj1->getPosition().x < obj2->getPosition().x; };
171  break;
172  case Direction::LEFT:
173  comparator = [](const T* obj1, const T* obj2) { return obj1->getPosition().x > obj2->getPosition().x; };
174  break;
175  }
176 
177  // STL function that returns an iterator to the minimal element using a given function to compare
178  return *std::min_element(objects.begin(), objects.end(), comparator);
179 }
